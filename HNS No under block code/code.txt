stock IsUnderBlock(id, ent)
{
	new Float:vEntOrigin[3];
	new Float:vPlayerOrigin[3];
	new Float:vPlayerMins[3];
	new Float:vEntMaxs[3];
	
	pev(id, pev_origin, vPlayerOrigin);
	pev(ent, pev_origin, vEntOrigin);
	pev(ent, pev_maxs, vEntMaxs);
	pev(id, pev_mins, vPlayerMins);
	
	if(vPlayerOrigin[2] + vPlayerMins[2] >= vEntOrigin[2] + vEntMaxs[2])
	{
		return false;
	}
	
	return true;
}
